CREATE TABLE  "USERINFO" 
   (	"NAME" VARCHAR2(20) NOT NULL ENABLE, 
	"USERNAME" VARCHAR2(20), 
	"DESIGNATION" VARCHAR2(20) NOT NULL ENABLE, 
	"PHONE" VARCHAR2(20) NOT NULL ENABLE, 
	"EMAIL" VARCHAR2(20) NOT NULL ENABLE, 
	"PWD1" VARCHAR2(20) NOT NULL ENABLE, 
	 CONSTRAINT "USERINFO_PK" PRIMARY KEY ("USERNAME") ENABLE
   ) ;
